17  package org.apache.commons.chain.impl;
18  
19  import org.apache.commons.chain.Chain;
20  import org.apache.commons.chain.Command;
21  import org.apache.commons.chain.Context;
22  
23  /**
24   * Implementation of {@link Command} that logs its identifier and
25   * and attempts to add a new {@link Command} to the {@link Chain}. This
26   * should cause an IllegalStateException if the {@link Chain} implementation
27   * subclasses {@code ChainBase}.
28   *
29   * @author Craig R. McClanahan
30   * @version $Revision$ $Date$
31   */
32  public class AddingCommand extends NonDelegatingCommand {
33  
34      // ------------------------------------------------------------ Constructor
35  
36      public AddingCommand() {
37          this("", null);
38      }
39  
40      /**
41       * Construct an instance that will log the specified identifier
42       *
43       * @param id identifier to log for this Command instance
44       * @param parent the parent Chain
45       */
46      public AddingCommand(String id, Chain<Context> parent) {
47          super(id);
48          this.parent = parent;
49      }
50  
51      /**
52       * The parent Chain
53       */
54      private Chain<Context> parent = null;
55  
56      // -------------------------------------------------------- Command Methods
57  
58      /**
59       * Execution method for this Command
60       *
61       * @param context The {@link Context} to be processed by this
62       *        {@link Command}
63       * @param chain the parent Chain
64       *
65       * @return {@code true} if the processing of this {@link Context}
66       *         has been completed, or {@code false} if the processing
67       *         of this {@link Context} should be delegated to a
68       *         subsequent {@link Command} in an enclosing {@link Chain}
69       *
70       * @throws Exception general purpose exception return
71       *         to indicate abnormal termination
72       * @throws IllegalArgumentException if {@code context}
73       *         is {@code null}
74       */
75      public boolean execute(Context context, Chain<Context> chain) throws Exception {
76          super.execute(context);
77          parent.addCommand(new NonDelegatingCommand("NEW")); // Should cause ISE
78          return true;
79      }
80  }
