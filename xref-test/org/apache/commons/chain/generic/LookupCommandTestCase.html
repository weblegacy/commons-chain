17  package org.apache.commons.chain.generic;
18  
19  import static org.junit.jupiter.api.Assertions.assertEquals;
20  import static org.junit.jupiter.api.Assertions.assertFalse;
21  import static org.junit.jupiter.api.Assertions.assertNotNull;
22  import static org.junit.jupiter.api.Assertions.assertTrue;
23  import static org.junit.jupiter.api.Assertions.fail;
24  
25  import org.apache.commons.chain.Catalog;
26  import org.apache.commons.chain.CatalogFactory;
27  import org.apache.commons.chain.Context;
28  import org.apache.commons.chain.impl.CatalogBase;
29  import org.apache.commons.chain.impl.ChainBase;
30  import org.apache.commons.chain.impl.ContextBase;
31  import org.apache.commons.chain.impl.DelegatingCommand;
32  import org.apache.commons.chain.impl.NonDelegatingCommand;
33  import org.junit.jupiter.api.AfterEach;
34  import org.junit.jupiter.api.BeforeEach;
35  import org.junit.jupiter.api.Test;
36  
37  /**
38   * Test case for the {@code LookupCommand} class.
39   *
40   * @author Sean Schofield
41   * @version $Revision$
42   */
43  public class LookupCommandTestCase {
44  
45      // ---------------------------------------------------- Instance Variables
46  
47      /**
48       * The instance of {@link Catalog} to use when looking up commands
49       */
50      protected Catalog<Context> catalog;
51  
52      /**
53       * The {@link LookupCommand} instance under test.
54       */
55      protected LookupCommand<Context> command;
56  
57      /**
58       * The {@link Context} instance on which to execute the chain.
59       */
60      protected Context context = null;
61  
62      // -------------------------------------------------- Overall Test Methods
63  
64      /**
65       * Set up instance variables required by this test case.
66       */
67      @BeforeEach
68      public void init() {
69          catalog = new CatalogBase<>();
70          CatalogFactory.getInstance().setCatalog(catalog);
71          command = new LookupCommand<>();
72          context = new ContextBase();
73      }
74  
75      /**
76       * Tear down instance variables required by this test case.
77       */
78      @AfterEach
79      public void tearDown() {
80          catalog = null;
81          CatalogFactory.clear();
82          command = null;
83          context = null;
84      }
85  
86      // ------------------------------------------------ Individual Test Methods
87  
88      /**
89       * Test ability to lookup and execute single non-delegating command
90       */
91      @Test
92      public void testExecuteMethodLookup_1a() {
93          // use default catalog
94          catalog.addCommand("foo", new NonDelegatingCommand("1a"));
95          command.setName("foo");
96  
97          try {
98              assertTrue(command.execute(context),
99                         "Command should return true");
100         } catch (Exception e) {
101             fail("Threw exception: " + e);
102         }
103         checkExecuteLog("1a");
104     }
105 
106     /**
107      * Test ability to lookup and execute a chain
108      */
109     @Test
110     public void testExecuteMethodLookup_1b() {
111         ChainBase<Context> chain = new ChainBase<>();
112         chain.addCommand(new DelegatingCommand("1b1"));
113         chain.addCommand(new DelegatingCommand("1b2"));
114         chain.addCommand(new NonDelegatingCommand("1b3"));
115 
116         // use default catalog
117         catalog.addCommand("foo", chain);
118         command.setName("foo");
119 
120         try {
121             assertTrue(command.execute(context),
122                        "Command should return true");
123         } catch (Exception e) {
124             fail("Threw exception: " + e);
125         }
126         checkExecuteLog("1b1/1b2/1b3");
127     }
128 
129     /**
130      * Test ability to lookup and execute single non-delegating command
131      * using the context
132      */
133     @Test
134     public void testExecuteMethodLookup_2a() {
135         // use default catalog
136         catalog.addCommand("foo", new NonDelegatingCommand("2a"));
137         command.setNameKey("nameKey");
138         context.put("nameKey", "foo");
139 
140         try {
141             assertTrue(command.execute(context),
142                        "Command should return true");
143         } catch (Exception e) {
144             fail("Threw exception: " + e);
145         }
146         checkExecuteLog("2a");
147     }
148 
149     /**
150      * Test ability to lookup and execute a chain using the context
151      */
152     @Test
153     public void testExecuteMethodLookup_2b() {
154         ChainBase<Context> chain = new ChainBase<>();
155         chain.addCommand(new DelegatingCommand("2b1"));
156         chain.addCommand(new DelegatingCommand("2b2"));
157         chain.addCommand(new NonDelegatingCommand("2b3"));
158 
159         // use default catalog
160         catalog.addCommand("foo", chain);
161         command.setNameKey("nameKey");
162         context.put("nameKey", "foo");
163 
164         try {
165             assertTrue(command.execute(context),
166                        "Command should return true");
167         } catch (Exception e) {
168             fail("Threw exception: " + e);
169         }
170         checkExecuteLog("2b1/2b2/2b3");
171     }
172 
173     /**
174      * Test ability to lookup and execute single non-delegating command, ignoring its result
175      */
176     @Test
177     public void testExecuteMethodLookup_3a() {
178         // use default catalog
179         catalog.addCommand("foo", new NonDelegatingCommand("3a"));
180         command.setIgnoreExecuteResult(true);
181         command.setName("foo");
182 
183         try {
184             assertFalse(command.execute(context),
185                         "Command should return false");
186         } catch (Exception e) {
187             fail("Threw exception: " + e);
188         }
189         checkExecuteLog("3a");
190     }
191 
192     // -------------------------------------------------------- Support Methods
193 
194     /**
195      * Verify the contents of the execution log
196      *
197      * @param expected the expected value
198      */
199     protected void checkExecuteLog(String expected) {
200         StringBuffer log = (StringBuffer) context.get("log");
201         assertNotNull(log, "Context failed to return log");
202         assertEquals(expected, log.toString(),
203                      "Context returned correct log");
204     }
205 }
