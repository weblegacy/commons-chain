17  package org.apache.commons.chain.web.jakarta;
18  
19  import org.apache.commons.chain.Catalog;
20  import org.apache.commons.chain.CatalogFactory;
21  import org.apache.commons.chain.config.ConfigParser;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import jakarta.servlet.ServletContext;
26  import jakarta.servlet.ServletContextEvent;
27  import jakarta.servlet.ServletContextListener;
28  import jakarta.servlet.ServletException;
29  
30  /**
31   * {@code ServletContextListener} that automatically
32   * scans chain configuration files in the current web application at
33   * startup time, and exposes the result in a {@link Catalog} under a
34   * specified servlet context attribute. The following <em>context</em> init
35   * parameters are utilized:
36   * <ul>
37   * <li><strong>org.apache.commons.chain.CONFIG_CLASS_RESOURCE</strong> -
38   *     comma-delimited list of chain configuration resources to be loaded
39   *     via {@code ClassLoader.getResource()} calls. If not specified,
40   *     no class loader resources will be loaded.</li>
41   * <li><strong>org.apache.commons.chain.CONFIG_WEB_RESOURCE</strong> -
42   *     comma-delimited list of chain configuration webapp resources
43   *     to be loaded. If not specified, no web application resources
44   *     will be loaded.</li>
45   * <li><strong>org.apache.commons.chain.CONFIG_ATTR</strong> -
46   *     Name of the servlet context attribute under which the
47   *     resulting {@link Catalog} will be created or updated.
48   *     If not specified, it is expected that parsed resources will
49   *     contain {@code &lt;catalog&gt;} elements (which will
50   *     cause registration of the created {@link Catalog}s into
51   *     the {@link CatalogFactory} for this application, and no
52   *     servet context attribute will be created.
53   *     <strong>NOTE</strong> - This parameter is deprecated.</li>
54   * <li><strong>org.apache.commons.chain.RULE_SET</strong> -
55   *     Fully qualified class name of a Digester {@code RuleSet}
56   *     implementation to use for parsing configuration resources (this
57   *     class must have a public zero-args constructor). If not defined,
58   *     the standard {@code RuleSet} implementation will be used.</li>
59   * </ul>
60   *
61   * <p>When a web application that has configured this listener is
62   * started, it will acquire the {@link Catalog} under the specified servlet
63   * context attribute key, creating a new one if there is none already there.
64   * This {@link Catalog} will then be populated by scanning configuration
65   * resources from the following sources (loaded in this order):</p>
66   * <ul>
67   * <li>Resources loaded from any {@code META-INF/chain-config.xml}
68   *     resource found in a JAR file in {@code /WEB-INF/lib}.</li>
69   * <li>Resources loaded from specified resource paths from the
70   *     webapp's class loader (via {@code ClassLoader.getResource()}).</li>
71   * <li>Resources loaded from specified resource paths in the web application
72   *     archive (via {@code ServetContext.getResource()}).</li>
73   * </ul>
74   *
75   * <p>If no attribute key is specified, on the other hand, parsed configuration
76   * resources are expected to contain {@code &lt;catalog&gt;} elements,
77   * and the catalogs will be registered with the {@link CatalogFactory}
78   * for this web application.</p>
79   *
80   * <p>This class requires Servlet 2.3 or later. If you are running on
81   * Servlet 2.2 system, consider using {@link ChainServlet} instead.
82   * Note that {@link ChainServlet} uses parameters of the
83   * same names, but they are <em>servlet</em> init parameters instead
84   * of <em>context</em> init parameters. Because of this, you can use
85   * both facilities in the same application, if desired.</p>
86   *
87   * @author Craig R. McClanahan
88   * @author Ted Husted
89   * @version $Revision$ $Date$
90   */
91  public class ChainListener implements ServletContextListener {
92  
93      // ------------------------------------------------------ Manifest Constants
94  
95      /**
96       * The name of the context init parameter containing the name of the
97       * servlet context attribute under which our resulting {@link Catalog}
98       * will be stored.
99       */
100     public static final String CONFIG_ATTR = ChainInit.CONFIG_ATTR;
101 
102     /**
103      * The name of the context init parameter containing a comma-delimited
104      * list of class loader resources to be scanned.
105      */
106     public static final String CONFIG_CLASS_RESOURCE =
107             ChainInit.CONFIG_CLASS_RESOURCE;
108 
109     /**
110      * The name of the context init parameter containing a comma-delimited
111      * list of web application resources to be scanned.
112      */
113     public static final String CONFIG_WEB_RESOURCE =
114             ChainInit.CONFIG_WEB_RESOURCE;
115 
116     /**
117      * The name of the context init parameter containing the fully
118      * qualified class name of the {@code RuleSet} implementation
119      * for configuring our {@link ConfigParser}.
120      */
121     public static final String RULE_SET = ChainInit.RULE_SET;
122 
123     // ------------------------------------------------------------ Constructors
124 
125     /**
126      * The Default-Constructor for this class.
127      */
128     public ChainListener() {
129     }
130 
131     // ------------------------------------------ ServletContextListener Methods
132 
133     /**
134      * Remove the configured {@link Catalog} from the servlet context
135      * attributes for this web application.
136      *
137      * @param event {@code ServletContextEvent} to be processed
138      */
139     @Override
140     public void contextDestroyed(ServletContextEvent event) {
141         final ServletContext context = event.getServletContext();
142         ChainInit.destroy(context, context.getInitParameter(CONFIG_ATTR));
143     }
144 
145     /**
146      * Scan the required chain configuration resources, assemble the
147      * configured chains into a {@link Catalog}, and expose it as a
148      * servlet context attribute under the specified key.
149      *
150      * @param event {@code ServletContextEvent} to be processed
151      */
152     @Override
153     public void contextInitialized(ServletContextEvent event) {
154         final Logger logger = LoggerFactory.getLogger(ChainListener.class);
155         logger.info("Initializing chain listener");
156         final ServletContext context = event.getServletContext();
157 
158         try {
159             ChainInit.initialize(context, context.getInitParameter(CONFIG_ATTR), logger, true);
160         } catch (ServletException e) {
161             throw new RuntimeException(e);
162         }
163     }
164 }
