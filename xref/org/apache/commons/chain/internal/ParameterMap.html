19  import java.util.ArrayList;
20  import java.util.Collection;
21  import java.util.Enumeration;
22  import java.util.HashSet;
23  import java.util.Iterator;
24  import java.util.List;
25  import java.util.Map;
26  import java.util.Objects;
27  import java.util.Set;
28  import java.util.function.Function;
29  import java.util.function.Supplier;

33  /**
34   * Implementation of {@code Map} for immutable parameters with a
35   * parameter-provider.
36   *
37   * @param <P> the type of the parameter-provider
38   * @param <T> the type of results supplied by this parameters
39   *
40   * @author Graff Stefan
41   * @since Chain 1.3
42   */
43  public class ParameterMap<P, T> implements Map<String, T> {
44  
45      /**
46       * The parameter-provider.
47       */
48      private final P parameter;
49  
50      /**
51       * Function to return the value of a parameter as an
52       * {@link String}, or {@code null} if no parameter of the
53       * given name exists.
54       */
55      private final Function<String, T> valueFunction;
56  
57      /**
58       * Supplier to return an {@link Enumeration} of {@link String}
59       * objects containing the names of the parameters contained
60       * in this object.
61       */
62      private final Supplier<Enumeration<String>> namesSupplier;
63  
64      /**
65       * The constructor for an immutable parameter-map.
66       *
67       * @param parameter the parameter-provider
68       * @param valueFunction Function to return the value of a parameter
69       * @param namesSupplier Supplier to return all names of the parameter
70       */
71      public ParameterMap(final P parameter, final Function<String, T> valueFunction,
72              final Supplier<Enumeration<String>> namesSupplier) {
73  
74          this.parameter = parameter;
75          this.valueFunction = valueFunction;
76          this.namesSupplier = namesSupplier;
77      }
78  
79      /**
80       * Removes all of the mappings from this parameter-map.
81       * The parameter-map will be empty after this call returns.
82       *
83       * @throws UnsupportedOperationException because it is an
84       *         immutable parameter-map
85       */
86      @Override
87      public void clear() {
88          throw new UnsupportedOperationException();
89      }
90  
91      /**
92       * Returns {@code true} if this parameter-map contains a mapping
93       * for the specified key.
94       *
95       * @param key The key whose presence in this parameter-map is to
96       *            be tested
97       *
98       * @return {@code true} if this parameter-map contains a mapping
99       *         for the specified key.
100      */
101     @Override
102     public boolean containsKey(Object key) {
103         return valueFunction.apply(key(key)) != null;
104     }
105 
106     /**
107      * Returns {@code true} if this parameter-map maps one or more keys
108      * to the specified value.
109      *
110      * @param value value whose presence in this parameter-map is to be
111      *        tested
112      *
113      * @return {@code true} if this parameter-map maps one or more keys
114      *         to the specified value
115      */
116     @Override
117     public boolean containsValue(Object value) {
118         final Enumeration<String> keys = namesSupplier.get();
119         while (keys.hasMoreElements()) {
120             final T next = valueFunction.apply(keys.nextElement());
121             if (value.equals(next)) {
122                 return true;
123             }
124         }
125         return false;
126     }
127 
128     /**
129      * Returns a {@link Set} view of the mappings contained in this
130      * parameter-map. The set is not backed by the parameter-map, so
131      * changes to the parameter-map are not reflected in the set,
132      * and vice-versa.
133      *
134      * @return a set view of the mappings contained in this
135      *         parameter-map
136      */
137     @Override
138     public Set<Map.Entry<String, T>> entrySet() {
139         return entrySet(false);
140     }
141 
142     /**
143      * Returns the value to which the specified key is mapped,
144      * or {@code null} if this parameter-map contains no mapping
145      * for the key.
146      *
147      * <p>A return value of {@code null} does not <i>necessarily</i>
148      * indicate that the parameter-map contains no mapping for the key;
149      * it's also possible that the parameter-map explicitly maps the key
150      * to {@code null}. The {@link #containsKey containsKey} operation
151      * may be used to distinguish these two cases.</p>
152      *
153      * @param key the key whose associated value is to be returned
154      *
155      * @return the value to which the specified key is mapped, or
156      *         {@code null} if this parameter-map contains no mapping for
157      *         the key
158      *
159      * @see #put(Object, Object)
160      */
161     @Override
162     public T get(Object key) {
163         return valueFunction.apply(key(key));
164     }
165 
166     /**
167      * Returns {@code true} if this parameter-map contains no
168      * key-value mappings.
169      *
170      * @return {@code true} if this parameter-map contains no
171      *         key-value mappings
172      */
173     @Override
174     public boolean isEmpty() {
175         return !namesSupplier.get().hasMoreElements();
176     }
177 
178     /**
179      * Returns a {@link Set} view of the keys contained in this
180      * parameter-map. The set is not backed by the parameter-map, so
181      * changes to the parameter-map are not reflected in the set, and
182      * vice-versa.
183      *
184      * @return a set view of the keys contained in this parameter-map
185      */
186     @Override
187     public Set<String> keySet() {
188         final Set<String> set = new HashSet<>();
189         final Enumeration<String> keys = namesSupplier.get();
190         while (keys.hasMoreElements()) {
191             set.add(keys.nextElement());
192         }
193         return set;
194     }
195 
196     /**
197      * Associates the specified value with the specified key in this
198      * parameter-map. If the parameter-map previously contained a
199      * mapping for the key, the old value is replaced.
200      *
201      * @param key key with which the specified value is to be associated
202      * @param value value to be associated with the specified key
203      *
204      * @return the previous value associated with {@code key}, or
205      *         {@code null} if there was no mapping for {@code key}.
206      *         (A {@code null} return can also indicate that the
207      *         parameter-map previously associated {@code null} with
208      *         {@code key.)
209      *
210      * @throws UnsupportedOperationException because it is an
211      *         immutable parameter-map
212      */
213     @Override
214     public T put(String key, T value) {
215         throw new UnsupportedOperationException();
216     }
217 
218     /**
219      * Copies all of the mappings from the specified map to this
220      * parameter-map. These mappings will replace any mappings that
221      * this parameter-map had for any of the keys currently in the
222      * specified map.
223      *
224      * @param map mappings to be stored in this parameter-map
225      *
226      * @throws NullPointerException if the specified map is null
227      * @throws UnsupportedOperationException because it is an
228      *         immutable parameter-map
229      */
230     @Override
231     public void putAll(Map<? extends String, ? extends T> map) {
232         throw new UnsupportedOperationException();
233     }
234 
235     /**
236      * Removes the mapping for the specified key from this
237      * parameter-map if present.
238      *
239      * @param key key whose mapping is to be removed from the
240      *        parameter-map
241      *
242      * @return the previous value associated with {@code key}, or
243      *         {@code null} if there was no mapping for {@code key}.
244      *         (A {@code null} return can also indicate that the
245      *         parameter-map previously associated {@code null} with
246      *         {@code key.)
247      *
248      * @throws UnsupportedOperationException because it is an
249      *         immutable parameter-map
250      */
251     @Override
252     public T remove(Object key) {
253         throw new UnsupportedOperationException();
254     }
255 
256     /**
257      * Returns the number of key-value mappings in this parameter-map.
258      *
259      * @return the number of key-value mappings in this parameter-map
260      */
261     @Override
262     public int size() {
263         int n = 0;
264         final Enumeration<String> keys = namesSupplier.get();
265         while (keys.hasMoreElements()) {
266             keys.nextElement();
267             n++;
268         }
269         return n;
270     }
271 
272     /**
273      * Returns a {@link Collection} view of the values contained in
274      * this parameter-map. The collection is not backed by the
275      * parameter-map, so changes to the parameter-map are not
276      * reflected in the collection, and vice-versa.
277      *
278      * @return a view of the values contained in this parameter-map
279      */
280     @Override
281     public Collection<T> values() {
282         final List<T> list = new ArrayList<>();
283         final Enumeration<String> keys = namesSupplier.get();
284         while (keys.hasMoreElements()) {
285             list.add(valueFunction.apply(keys.nextElement()));
286         }
287         return list;
288     }
289 
290     /**
291      * Returns the hash code value for this parameter-map. The
292      * hash code of a parameter-map is defined to be the sum of
293      * the hash codes of each entry in the parameter-map's
294      * {@code entrySet()} view. This ensures that {@code m1.equals(m2)}
295      * implies that {@code m1.hashCode()==m2.hashCode()} for any two
296      * parameter-maps {@code m1} and {@code m2}, as required by the
297      * general contract of {@link Object#hashCode}.
298      *
299      * @implSpec
300      * This implementation calls the {@code hashCode()} from the
301      * parameter-provider.
302      *
303      * @return the hash code value for this parameter-map
304      */
305     @Override
306     public int hashCode() {
307         return getParameter().hashCode();
308     }
309 
310     /**
311      * Compares the specified object with this parameter-map for equality.
312      * Returns {@code true} if the given object is also a parameter-map
313      * and the two parameter-maps represent the same mappings. More formally,
314      * two parameter-maps {@code m1} and {@code m2} represent the same
315      * mappings if {@code m1.entrySet().equals(m2.entrySet())}.
316      *
317      * @implSpec
318      * This implementation first checks if the specified object is this
319      * parameter-map; if so it returns {@code true}. Then, it checks if
320      * the specified object is the identical class this parameter-map; if
321      * not, it returns {@code false}. If so, it calls the {@code equals()}
322      * from the parameter-provider and returns its return-code.
323      *
324      * @param obj object to be compared for equality with this
325      *        parameter-map
326      *
327      * @return {@code true} if the specified object is equal to this
328      *         parameter-map
329      */
330     @Override
331     public boolean equals(Object obj) {
332         if (this == obj) {
333             return true;
334         }
335         if (obj == null || getClass() != obj.getClass()) {
336             return false;
337         }
338         final ParameterMap<?, ?> other = (ParameterMap<?, ?>) obj;
339         return Objects.equals(getParameter(), other.getParameter());
340     }
341 
342     /**
343      * Returns a string representation of this parameter-map.
344      * The string representation consists of a list of key-value
345      * mappings in the order returned by the parameter-map's
346      * {@code entrySet} view's iterator, enclosed in braces
347      * ({@code "{}"}). Adjacent mappings are separated by the
348      * characters {@code ", "} (comma and space). Each key-value
349      * mapping is rendered as the key followed by an equals sign
350      * ({@code "="}) followed by the associated value. Keys and
351      * values are converted to strings as by
352      * {@link String#valueOf(Object)}.
353      *
354      * @return a string representation of this parameter-map
355      */
356     public String toString() {
357         final Iterator<Entry<String, T>> entries = entrySet().iterator();
358         if (entries.hasNext()) {
359             return "{}";
360         }
361 
362         final StringBuilder sb = new StringBuilder();
363         sb.append('{');
364         for (;;) {
365             final Entry<String, T> entrie = entries.next();
366             final String key = entrie.getKey();
367             final T value = entrie.getValue();
368             sb
369                 .append(key)
370                 .append('=')
371                 .append(value);
372 
373             if (entries.hasNext()) {
374                 sb.append(',').append(' ');
375             } else {
376                 return sb.append('}').toString();
377             }
378         }
379     }

381     /**
382      * Returns the parameter-class.
383      *
384      * @return the parameter-class
385      */
386     protected P getParameter() {
387         return parameter;
388     }
389 
390     /**
391      * Returns the Function to return the value of a parameter as an
392      * {@link String}, or {@code null} if no parameter of the given
393      * name exists.
394      *
395      * @return the Function to return the value
396      */
397     protected  Function<String, T> getValueFunction() {
398         return valueFunction;
399     }
400 
401     /**
402      * Returns the Supplier to return an {@link Enumeration} of
403      * {@link String} objects containing the names of the parameters
404      * contained in this object.
405      *
406      * @return Supplier to return an {@link Enumeration} with all
407      *         names of the parameters
408      */
409     protected Supplier<Enumeration<String>> getNamesSupplier() {
410         return namesSupplier;
411     }
412 
413     /**
414      * Returns a {@link Set} view of the mappings contained
415      * in this map.
416      *
417      * @param modifiable Whether the entries should allow
418      *                   modification or not
419      *
420      * @return a set view of the mappings contained in this map
421      */
422     protected Set<Map.Entry<String, T>> entrySet(boolean modifiable) {
423         final Set<Map.Entry<String, T>> set = new HashSet<>();
424         final Enumeration<String> keys = namesSupplier.get();
425         while (keys.hasMoreElements()) {
426             final String key = keys.nextElement();
427             set.add(new MapEntry<>(key, valueFunction.apply(key), modifiable));
428         }
429         return set;
430     }
431 
432     /**
433      * Converts the {@code key} to a {@link String}.
434      *
435      * @param key the key
436      *
437      * @return the key as {@link String}
438      */
439     protected static String key(Object key) {
440         if (key == null) {
441             throw new IllegalArgumentException();
442         } else if (key instanceof String) {
443             return (String) key;
444         } else {
445             return key.toString();
446         }
447     }
