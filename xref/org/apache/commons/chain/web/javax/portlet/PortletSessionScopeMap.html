17  package org.apache.commons.chain.web.javax.portlet;
18  
19  import javax.portlet.PortletRequest;
20  import javax.portlet.PortletSession;
21  
22  import org.apache.commons.chain.web.AbstractSessionScopeMap;
23  import org.apache.commons.chain.web.MutableParameterMap;
24  
25  /**
26   * Private implementation of {@code Map} for portlet session attributes.
27   *
28   * @author Craig R. McClanahan
29   * @version $Revision$ $Date$
30   */
31  final class PortletSessionScopeMap extends AbstractSessionScopeMap<PortletSession, PortletRequest> {
32  
33      /**
34       * The constructor for the portlet session attributes.
35       *
36       * @param request the portlet-request.
37       */
38      PortletSessionScopeMap(PortletRequest request) {
39          super(request);
40      }
41  
42      /**
43       * Returns the current portlet session or, if there is no current session and
44       * the given flag is {@code true}, creates one and returns the new session.
45       *
46       * <p>If the given flag is {@code false} and there is no current portlet
47       * session, this method returns {@code null}.</p>
48       *
49       * <p>Creating a new portlet session will result in creating a new
50       * {@code HttpSession} on which the portlet session is based.</p>
51       *
52       * @param create {@code true} to create a new session, {@code false} to return
53       *               {@code null} if there is no current session
54       *
55       * @return the portlet session
56       */
57      @Override
58      protected PortletSession getSession(final boolean create) {
59          return getRequest().getPortletSession(create);
60      }
61  
62      /**
63       * Creates a new mutable-parameter-map to access the session-attributes.
64       *
65       * @return a new mutable-parameter-map to access the session-attributes
66       */
67      @Override
68      protected MutableParameterMap<PortletSession, Object> createParameterMap() {
69          return new MutableParameterMap<>(getSession(), getSession()::getAttribute,
70                  () -> getSession().getAttributeNames(PortletSession.PORTLET_SCOPE),
71                  getSession()::removeAttribute, getSession()::setAttribute);
72      }
73  }
